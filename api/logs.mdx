## Endpoint

```
GET /api/projects/{projectId}/logs
```

Retrieve detailed logs of all API requests made to your project.

## Authentication

Requires dashboard authentication (user session), not API key authentication.

## Query Parameters

<ParamField query="page" type="number">
  Page number for pagination
  
  Default: `1`
</ParamField>

<ParamField query="per_page" type="number">
  Results per page
  
  Default: `50`
  
  Max: `100`
</ParamField>

<ParamField query="time_range" type="string">
  Time range to query
  
  Options:
  - `1h` - Last hour
  - `24h` - Last 24 hours (default)
  - `7d` - Last 7 days
  - `30d` - Last 30 days
  - `all` - All time
</ParamField>

<ParamField query="status" type="string">
  Filter by request status
  
  Options:
  - `all` - All requests (default)
  - `success` - Successful requests
  - `error` - Failed requests
  - `filtered` - Content filtered
  - `blocked` - Security blocked
  - `rate_limited` - Rate limited
</ParamField>

<ParamField query="model" type="string">
  Filter by model name
  
  Example: `gpt-4o`, `gemini-2.0-flash`
</ParamField>

<ParamField query="api_key_id" type="string">
  Filter by API key ID
</ParamField>

<ParamField query="environment" type="string">
  Filter by environment
  
  Options: `production`, `test`
  
  Default: `production`
</ParamField>

<ParamField query="search" type="string">
  Search in request content and error messages
</ParamField>

### Using TypeScript

```typescript
const projectId = 'proj_123';

async function getLogs(options: {
  page?: number;
  status?: string;
  timeRange?: string;
  apiKeyId?: string;
}) {
  const params = new URLSearchParams({
    page: options.page?.toString() || '1',
    per_page: '50',
    time_range: options.timeRange || '24h',
    ...(options.status && { status: options.status }),
    ...(options.apiKeyId && { api_key_id: options.apiKeyId })
  });
  
  const response = await fetch(
    `https://api.cencori.com/api/projects/${projectId}/logs?${params}`,
    {
      headers: { 'Cookie': `session=${sessionToken}` }
    }
  );
  
  return await response.json();
}

// Get errors from last week
const errors = await getLogs({ status: 'error', timeRange: '7d' });
console.log(`Found ${errors.requests.length} errors`);
```

## Use Cases

### Debug Failed Requests

```typescript
async function debugFailures() {
  const logs = await getLogs({ status: 'error', timeRange: '24h' });
  
  // Group by error message
  const errorGroups: Record<string, any[]> = {};
  
  for (const log of logs.requests) {
    const key = log.error_message || 'Unknown error';
    if (!errorGroups[key]) errorGroups[key] = [];
    errorGroups[key].push(log);
  }
  
  // Print error summary
  console.log('Error Summary:');
  for (const [error, occurrences] of Object.entries(errorGroups)) {
    console.log(`  ${error}: ${occurrences.length} times`);
  }
}
```

### Monitor Security Incidents

```typescript
async function checkSecurityIncidents() {
  const logs = await getLogs({ status: 'filtered', timeRange: '24h' });
  
  console.log(`Total filtered requests: ${logs.requests.length}`);
  
  // Analyze filtered reasons
  const reasons: Record<string, number> = {};
  
  for (const log of logs.requests) {
    for (const reason of log.filtered_reasons || []) {
      reasons[reason] = (reasons[reason] || 0) + 1;
    }
  }
  
  console.log('\nFiltered by:');
  for (const [reason, count] of Object.entries(reasons)) {
    console.log(`  ${reason}: ${count}`);
  }
}
```

### Track Costs by API Key

```typescript
async function costByApiKey() {
  const logs = await getLogs({ timeRange: '30d' });
  
  const costs: Record<string, { total: number; requests: number }> = {};
  
  for (const log of logs.requests) {
    const key = log.api_key_name;
    if (!costs[key]) costs[key] = { total: 0, requests: 0 };
    costs[key].total += log.cost_usd;
    costs[key].requests++;
  }
  
  console.log('Cost by API Key:');
  for (const [key, data] of Object.entries(costs)) {
    console.log(`  ${key}: $${data.total.toFixed(2)} (${data.requests} requests)`);
  }
}
```

### Analyze Performance

```typescript
async function analyzeLatency() {
  const logs = await getLogs({ timeRange: '24h' });
  
  const latencies = logs.requests
    .filter(r => r.status === 'success')
    .map(r => r.latency_ms)
    .sort((a, b) => a - b);
  
  const avg = latencies.reduce((a, b) => a + b, 0) / latencies.length;
  const p50 = latencies[Math.floor(latencies.length * 0.5)];
  const p95 = latencies[Math.floor(latencies.length * 0.95)];
  const p99 = latencies[Math.floor(latencies.length * 0.99)];
  
  console.log('Latency Analysis:');
  console.log(`  Average: ${avg.toFixed(0)}ms`);
  console.log(`  P50: ${p50}ms`);
  console.log(`  P95: ${p95}ms`);
  console.log(`  P99: ${p99}ms`);
}
```

### Build Monitoring Dashboard

```typescript
import { useState, useEffect } from 'react';

function LogsDashboard({ projectId }: { projectId: string }) {
  const [logs, setLogs] = useState(null);
  const [filter, setFilter] = useState({ status: 'all', timeRange: '24h' });
  
  useEffect(() => {
    const params = new URLSearchParams(filter);
    fetch(`/api/projects/${projectId}/logs?${params}`)
      .then(r => r.json())
      .then(setLogs);
  }, [projectId, filter]);
  
  if (!logs) return <div>Loading...</div>;
  
  return (
    <div>
      <h2>Request Logs</h2>
      
      <select onChange={e => setFilter({...filter, status: e.target.value})}>
        <option value="all">All Requests</option>
        <option value="success">Success</option>
        <option value="error">Errors</option>
        <option value="filtered">Filtered</option>
      </select>
      
      <table>
        <thead>
          <tr>
            <th>Time</th>
            <th>Status</th>
            <th>Model</th>
            <th>Tokens</th>
            <th>Cost</th>
            <th>Latency</th>
          </tr>
        </thead>
        <tbody>
          {logs.requests.map(log => (
            <tr key={log.id}>
              <td>{new Date(log.created_at).toLocaleString()}</td>
              <td>{log.status}</td>
              <td>{log.model}</td>
              <td>{log.total_tokens}</td>
              <td>${log.cost_usd.toFixed(4)}</td>
              <td>{log.latency_ms}ms</td>
            </tr>
          ))}
        </tbody>
      </table>
      
      <Pagination data={logs.pagination} />
    </div>
  );
}
```

## Streaming Logs

For real-time log monitoring:

```
GET /api/projects/{projectId}/logs/stream
```

Returns Server-Sent Events (SSE) stream:

```typescript
const eventSource = new EventSource(
  `/api/projects/${projectId}/logs/stream`,
  { withCredentials: true }
);

eventSource.onmessage = (event) => {
  const log = JSON.parse(event.data);
  console.log('New request:', log);
  
  if (log.status === 'error') {
    alert(`Error: ${log.error_message}`);
  }
};
```

## Best Practices

1. **Use time ranges efficiently**
   - Query recent data with `24h` or `7d`
   - Avoid `all` for large datasets

2. **Filter before processing**
   - Use API filters instead of client-side filtering
   - Reduces data transfer and processing

3. **Implement pagination**
   - Don't fetch all logs at once
   - Process in batches

4. **Monitor regularly**
   - Set up automated checks
   - Alert on anomalies

5. **Export for compliance**
   - Download logs monthly
   - Store for audit purposes

## Privacy & Compliance

- Logs may contain user content
- Follow data protection regulations (GDPR, CCPA)
- Use PII detection and redaction
- Configure data retention policies
- Export logs before retention expires
