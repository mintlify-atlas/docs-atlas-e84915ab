# Streaming Responses

Stream AI responses token-by-token as they're generated. Perfect for chat interfaces, real-time applications, and improving perceived latency.

## Why Streaming?

- **Better UX**: Users see responses appear immediately
- **Lower latency**: Display first tokens before full completion
- **Progress feedback**: Show that work is happening
- **Large responses**: Handle long outputs efficiently

## Basic Streaming

### Chat Stream

```typescript
import { Cencori } from 'cencori';

const cencori = new Cencori({ apiKey: process.env.CENCORI_API_KEY });

for await (const chunk of cencori.ai.chatStream({
  model: 'gpt-4o',
  messages: [
    { role: 'user', content: 'Write a short story about AI' }
  ]
})) {
  process.stdout.write(chunk.delta);
}
```

See `packages/sdk/src/ai/index.ts:166` for the `chatStream()` implementation.

### Complete Example

```typescript
import { Cencori, StreamChunk } from 'cencori';

const cencori = new Cencori({ apiKey: process.env.CENCORI_API_KEY });

const stream = cencori.ai.chatStream({
  model: 'gpt-4o',
  messages: [
    { role: 'system', content: 'You are a helpful assistant.' },
    { role: 'user', content: 'Explain TypeScript generics' }
  ],
  temperature: 0.7
});

let fullContent = '';

for await (const chunk of stream) {
  fullContent += chunk.delta;
  process.stdout.write(chunk.delta);
  
  // Check for completion
  if (chunk.finish_reason === 'stop') {
    console.log('\n\nStream complete!');
    break;
  }
}

console.log('Full response:', fullContent);
```

## Stream Chunk Format

### StreamChunk Type

```typescript
interface StreamChunk {
  delta: string;                 // Text delta for this chunk
  finish_reason?: 'stop' | 'length' | 'content_filter' | 'tool_calls' | 'error';
  toolCalls?: ToolCall[];        // Tool calls (if any)
  error?: string;                // Error message (if error occurred)
}
```

See `packages/sdk/src/ai/index.ts:81` for the `StreamChunk` definition.

### Handling Finish Reasons

```typescript
for await (const chunk of cencori.ai.chatStream({ model: 'gpt-4o', messages })) {
  process.stdout.write(chunk.delta);
  
  if (chunk.finish_reason) {
    switch (chunk.finish_reason) {
      case 'stop':
        console.log('\n[Completed normally]');
        break;
      case 'length':
        console.log('\n[Stopped: Max tokens reached]');
        break;
      case 'content_filter':
        console.log('\n[Stopped: Content filtered]');
        break;
      case 'tool_calls':
        console.log('\n[Waiting for tool execution]');
        console.log('Tool calls:', chunk.toolCalls);
        break;
      case 'error':
        console.error('\n[Error]:', chunk.error);
        break;
    }
  }
}
```

## Framework Integration

### Next.js API Route

Stream responses to the client:

```typescript app/api/chat/route.ts
import { Cencori } from 'cencori';
import { NextRequest } from 'next/server';

const cencori = new Cencori({ apiKey: process.env.CENCORI_API_KEY });

export async function POST(req: NextRequest) {
  const { messages } = await req.json();

  const stream = new ReadableStream({
    async start(controller) {
      const encoder = new TextEncoder();

      try {
        for await (const chunk of cencori.ai.chatStream({
          model: 'gpt-4o',
          messages
        })) {
          controller.enqueue(encoder.encode(`data: ${JSON.stringify(chunk)}\n\n`));
        }
        controller.enqueue(encoder.encode('data: [DONE]\n\n'));
      } catch (error) {
        controller.error(error);
      } finally {
        controller.close();
      }
    }
  });

  return new Response(stream, {
    headers: {
      'Content-Type': 'text/event-stream',
      'Cache-Control': 'no-cache',
      'Connection': 'keep-alive'
    }
  });
}
```

### React Client

Consume the stream in React:

```tsx components/Chat.tsx
import { useState } from 'react';

export function Chat() {
  const [messages, setMessages] = useState([]);
  const [input, setInput] = useState('');
  const [streaming, setStreaming] = useState(false);

  async function sendMessage() {
    setStreaming(true);
    
    const response = await fetch('/api/chat', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ messages: [...messages, { role: 'user', content: input }] })
    });

    const reader = response.body.getReader();
    const decoder = new TextDecoder();
    let content = '';

    while (true) {
      const { done, value } = await reader.read();
      if (done) break;

      const chunk = decoder.decode(value);
      const lines = chunk.split('\n').filter(line => line.trim().startsWith('data:'));

      for (const line of lines) {
        const data = line.replace('data: ', '');
        if (data === '[DONE]') continue;

        const parsed = JSON.parse(data);
        content += parsed.delta;
        
        // Update UI with streaming content
        setMessages(prev => [
          ...prev.slice(0, -1),
          { role: 'assistant', content }
        ]);
      }
    }

    setStreaming(false);
  }

  return (
    <div>
      {/* Chat UI */}
    </div>
  );
}
```

## RAG Streaming

Stream responses with automatic context retrieval:

```typescript
for await (const chunk of cencori.ai.ragStream({
  model: 'gpt-4o',
  messages: [{ role: 'user', content: 'What are our company policies?' }],
  namespace: 'company-docs',
  limit: 5
})) {
  if (chunk.type === 'sources') {
    console.log('Retrieved sources:', chunk.sources?.length);
    chunk.sources?.forEach(source => {
      console.log(`- ${source.content.slice(0, 100)}... (similarity: ${source.similarity})`);
    });
  } else if (chunk.type === 'content') {
    process.stdout.write(chunk.delta);
  }
}
```

## Advanced Patterns

### Accumulating Stream Content

```typescript
let fullContent = '';
let toolCalls: ToolCall[] = [];

for await (const chunk of cencori.ai.chatStream({ model: 'gpt-4o', messages })) {
  fullContent += chunk.delta;
  
  if (chunk.toolCalls) {
    toolCalls.push(...chunk.toolCalls);
  }
  
  process.stdout.write(chunk.delta);
}

console.log('\n\nFull response:', fullContent);
if (toolCalls.length > 0) {
  console.log('Tool calls:', toolCalls);
}
```

### Stream with Timeout

```typescript
const TIMEOUT_MS = 30000; // 30 seconds

async function streamWithTimeout() {
  const timeoutPromise = new Promise((_, reject) => 
    setTimeout(() => reject(new Error('Stream timeout')), TIMEOUT_MS)
  );

  const streamPromise = (async () => {
    for await (const chunk of cencori.ai.chatStream({ model: 'gpt-4o', messages })) {
      process.stdout.write(chunk.delta);
    }
  })();

  await Promise.race([streamPromise, timeoutPromise]);
}
```

### Error Recovery

```typescript
import { CencoriError } from 'cencori';

try {
  for await (const chunk of cencori.ai.chatStream({ model: 'gpt-4o', messages })) {
    if (chunk.error) {
      console.error('Stream error:', chunk.error);
      break;
    }
    process.stdout.write(chunk.delta);
  }
} catch (error) {
  if (error instanceof CencoriError) {
    console.error('API Error:', error.message, error.statusCode);
  } else {
    console.error('Stream failed:', error);
  }
}
```

## Server-Sent Events (SSE)

Stream to browsers using SSE:

```typescript
import express from 'express';
import { Cencori } from 'cencori';

const app = express();
const cencori = new Cencori({ apiKey: process.env.CENCORI_API_KEY });

app.post('/api/chat/stream', async (req, res) => {
  res.setHeader('Content-Type', 'text/event-stream');
  res.setHeader('Cache-Control', 'no-cache');
  res.setHeader('Connection', 'keep-alive');

  const { messages } = req.body;

  try {
    for await (const chunk of cencori.ai.chatStream({
      model: 'gpt-4o',
      messages
    })) {
      res.write(`data: ${JSON.stringify(chunk)}\n\n`);
    }
    res.write('data: [DONE]\n\n');
  } catch (error) {
    res.write(`data: ${JSON.stringify({ error: error.message })}\n\n`);
  } finally {
    res.end();
  }
});
```

## Performance Tips

### Buffering

Buffer small chunks for smoother rendering:

```typescript
const BUFFER_SIZE = 5; // characters
let buffer = '';

for await (const chunk of cencori.ai.chatStream({ model: 'gpt-4o', messages })) {
  buffer += chunk.delta;
  
  if (buffer.length >= BUFFER_SIZE) {
    process.stdout.write(buffer);
    buffer = '';
  }
}

// Flush remaining buffer
if (buffer.length > 0) {
  process.stdout.write(buffer);
}
```

### Abort Controller

Cancel streams early:

```typescript
const controller = new AbortController();

// Cancel after 10 seconds
setTimeout(() => controller.abort(), 10000);

try {
  for await (const chunk of cencori.ai.chatStream({ model: 'gpt-4o', messages })) {
    if (controller.signal.aborted) {
      console.log('\nStream cancelled');
      break;
    }
    process.stdout.write(chunk.delta);
  }
} catch (error) {
  if (error.name === 'AbortError') {
    console.log('Stream was aborted');
  }
}
```

## Complete Example

Here's a production-ready streaming chat implementation:

```typescript
import { Cencori, StreamChunk, CencoriError, ChatMessage } from 'cencori';

const cencori = new Cencori({ apiKey: process.env.CENCORI_API_KEY });

interface StreamOptions {
  onChunk?: (chunk: StreamChunk) => void;
  onComplete?: (fullContent: string) => void;
  onError?: (error: Error) => void;
  timeout?: number;
}

async function streamChat(
  messages: ChatMessage[],
  options: StreamOptions = {}
) {
  const { onChunk, onComplete, onError, timeout = 60000 } = options;
  
  let fullContent = '';
  let isComplete = false;

  const timeoutPromise = new Promise<never>((_, reject) => 
    setTimeout(() => reject(new Error('Stream timeout')), timeout)
  );

  const streamPromise = (async () => {
    try {
      for await (const chunk of cencori.ai.chatStream({
        model: 'gpt-4o',
        messages,
        temperature: 0.7
      })) {
        if (chunk.error) {
          throw new Error(chunk.error);
        }

        fullContent += chunk.delta;
        onChunk?.(chunk);

        if (chunk.finish_reason === 'stop') {
          isComplete = true;
          onComplete?.(fullContent);
          break;
        }
      }
    } catch (error) {
      onError?.(error instanceof Error ? error : new Error(String(error)));
      throw error;
    }
  })();

  try {
    await Promise.race([streamPromise, timeoutPromise]);
  } catch (error) {
    if (!isComplete) {
      onError?.(error instanceof Error ? error : new Error(String(error)));
    }
    throw error;
  }

  return fullContent;
}

// Usage
streamChat(
  [{ role: 'user', content: 'Tell me a story' }],
  {
    onChunk: (chunk) => process.stdout.write(chunk.delta),
    onComplete: (content) => console.log('\n\nDone! Length:', content.length),
    onError: (error) => console.error('Error:', error.message)
  }
);
```
